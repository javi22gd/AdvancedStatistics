or <- function(x, y){
  p <- c()
  r <- c()
  for(i in seq(1, 4, 2)){
    p <- c(p, ((table(x, y)[i+1] / ((table(x, y)[i]) + table(x, y)[i+1]))*100) / ((table(x, y)[i] / ((table(x, y)[i]) + table(x, y)[i+1]))*100))
  }
  return(p[2]/p[1])
}
